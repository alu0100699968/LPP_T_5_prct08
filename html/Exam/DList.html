def pushHead(v)
  if v.kind_of?(Array)
    for i in 0..v.size-1
        aux = @head
        nodo=Node.new(v[i], aux, nil)
        @head=nodo
        aux[:prev]=@head
    end
  else
    aux = @head
    nodo=Node.new(v, aux, nil)
    @head=nodo
    aux[:prev]=@head
  end
end

def to_is
  aux=@head

  end
  while aux[:prev] do
